        async function generateImages() {
            const prompt = document.getElementById('prompt').value;

            const provider = document.getElementById('apiProvider').value;
            const randomShuffle = document.getElementById('randomShuffle').checked;

        async function generateSingleImage(prompt, provider, index) {
            try {
                let imageUrl;
                
                switch (provider) {
                    case 'pollinations':
                    default:
                        imageUrl = `https://image.pollinations.ai/prompt/${encodeURIComponent(prompt)}?width=1024&height=1024&seed=${Math.floor(Math.random() * 10000) + index}`;
                        break;
                        
                    case 'deepai':
                    case 'stablediffusion':
                    case 'prodia':
                    case 'playground':
                    case 'kandinsky':
                    case 'nightcafe':
                    case 'dreamlike':
                    case 'aismurf':
                    case 'flux':
                    case 'nsfw':
                        // For other APIs, use Pollinations as fallback for this static version
                        imageUrl = `https://image.pollinations.ai/prompt/${encodeURIComponent(prompt)}?width=1024&height=1024&seed=${Math.floor(Math.random() * 10000) + index}`;
                        break;
                }
                
                return {
                    url: imageUrl,
                    provider: provider.charAt(0).toUpperCase() + provider.slice(1),
                    index: index,
                    prompt: prompt
                };
            } catch (error) {
                console.error(`Failed to generate image ${index}:`, error);
                return null;
            }
        }

        function createSimpleAudio(text) {
            // Create a minimal audio file
            const sampleRate = 44100;
            const duration = Math.min(text.length / 10, 30);
            const numSamples = sampleRate * duration;
            const arrayBuffer = new ArrayBuffer(44 + numSamples * 2);
            const view = new DataView(arrayBuffer);
            
            // WAV header
            view.setUint32(0, 0x46464952, false); // "RIFF"
            view.setUint32(4, 36 + numSamples * 2, true);
            view.setUint32(8, 0x45564157, false); // "WAVE"
            view.setUint32(12, 0x20746D66, false); // "fmt "
            view.setUint32(16, 16, true);
            view.setUint16(20, 1, true);
            view.setUint16(22, 1, true);
            view.setUint32(24, sampleRate, true);
            view.setUint32(28, sampleRate * 2, true);
            view.setUint16(32, 2, true);
            view.setUint16(34, 16, true);
            view.setUint32(36, 0x61746164, false); // "data"
            view.setUint32(40, numSamples * 2, true);
            
            return new Blob([arrayBuffer], { type: 'audio/wav' });
        }

                    <div class="loading-subtext">Searching for images across the web...</div>
                </div>
            `;

            try {
                const response = await fetch(`${BACKEND_URL}/api/scrape`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        keyword: keyword,
                        num_images: numImages,
                        quality: quality,
                        safe_search_off: true
                    })
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const data = await response.json();
                
                if (data.error) {
                    throw new Error(data.error);
                }

                currentScrapeTaskId = data.task_id;
                
                // Start polling for status
                scrapePollingInterval = setInterval(checkScrapeStatus, 2000);
                
            } catch (error) {
                console.error('Scraping error:', error);
                container.innerHTML = `
                    <div class="empty-state">
                        <div style="color: #FCA5A5; font-size: 2rem; margin-bottom: 15px;">‚ö†Ô∏è</div>
                        <div style="color: #FCA5A5; font-size: 1.2rem; margin-bottom: 10px;">Scraping Failed</div>
                        <div style="color: #94A3B8; font-size: 0.9rem;">${error.message}</div>
                    </div>
                `;
                scrapBtn.disabled = false;
                scrapBtn.innerHTML = '<span class="btn-icon">üîç</span><span>Start Web Scraping</span>';
            }

        // Check scraping status
        async function checkScrapeStatus() {
            if (!currentScrapeTaskId) return;

            try {
                const response = await fetch(`${BACKEND_URL}/api/status/${currentScrapeTaskId}`);
                const data = await response.json();
                
                updateScrapeProgress(data);
                
                if (data.status === 'completed') {
                    clearInterval(scrapePollingInterval);
                    scrapePollingInterval = null;
                    allScrapedImages = data.images || [];
                    displayScrapedImages(data);
                    document.getElementById('scrapBtn').disabled = false;
                    document.getElementById('scrapBtn').innerHTML = '<span class="btn-icon">üîç</span><span>Start Web Scraping</span>';
                } else if (data.status === 'error') {
                    clearInterval(scrapePollingInterval);
                    scrapePollingInterval = null;
                    showScrapError(data.message || 'Web scraping failed');
                    document.getElementById('scrapBtn').disabled = false;
                    document.getElementById('scrapBtn').innerHTML = '<span class="btn-icon">üîç</span><span>Start Web Scraping</span>';
                }
                
            } catch (error) {
                console.error('Status check error:', error);
                clearInterval(scrapePollingInterval);
                scrapePollingInterval = null;
                showScrapError('Failed to check scraping status');
                document.getElementById('scrapBtn').disabled = false;
                document.getElementById('scrapBtn').innerHTML = '<span class="btn-icon">üîç</span><span>Start Web Scraping</span>';
            }
        }

        // Download single scraped image
        async function downloadSingleImage(imageIndex) {
            if (!currentScrapeTaskId) return;
            
            try {
                const imageUrl = `${BACKEND_URL}/api/image/${currentScrapeTaskId}/${imageIndex}`;
                const response = await fetch(imageUrl);
                
                if (!response.ok) throw new Error(`Failed to download image: HTTP ${response.status}`);
                
                const blob = await response.blob();
                const url = URL.createObjectURL(blob);
                const filename = allScrapedImages[imageIndex]?.filename || `web_image_${imageIndex + 1}.jpg`;
                
                const link = document.createElement('a');
                link.href = url;
                link.download = filename;
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
                
            } catch (error) {
                console.error('Download error:', error);
                alert('Failed to download image');
            }
        }

        function generatePassword() {
            const length = parseInt(document.getElementById('passwordLength').value);
            const count = parseInt(document.getElementById('passwordCount').value);
            const includeUpper = document.getElementById('includeUppercase').checked;
            const includeLower = document.getElementById('includeLowercase').checked;
            const includeNumbers = document.getElementById('includeNumbers').checked;
            const includeSymbols = document.getElementById('includeSymbols').checked;
            
            let charset = '';
            if (includeUpper) charset += 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
            if (includeLower) charset += 'abcdefghijklmnopqrstuvwxyz';
            if (includeNumbers) charset += '0123456789';
            if (includeSymbols) charset += '!@#$%^&*()_+-=[]{}|;:,.<>?';
            
            if (charset === '') {
                alert('Please select at least one character type.');
                return;
            }
            
            const passwords = [];
            for (let i = 0; i < count; i++) {
                let password = '';
                for (let j = 0; j < length; j++) {
                    password += charset.charAt(Math.floor(Math.random() * charset.length));
                }
                passwords.push(password);
            }
            
            document.getElementById('passwordResult').innerHTML = `
                <div style="background: var(--glass-bg); padding: 20px; border-radius: 15px;">
                    <h4 style="color: var(--accent-purple); margin-bottom: 15px;">Generated Passwords:</h4>
                    ${passwords.map(pwd => `
                        <div style="display: flex; align-items: center; margin-bottom: 10px; background: rgba(0,0,0,0.3); padding: 10px; border-radius: 8px;">
                            <code style="flex: 1; color: var(--text-light); word-break: break-all;">${pwd}</code>
                            <button onclick="navigator.clipboard.writeText('${pwd}')" class="download-btn" style="margin-left: 10px; padding: 6px 12px;">Copy</button>
                        </div>
                    `).join('')}
                </div>
            `;
        }

        async function sendMessage() {
            const input = document.getElementById('chatInput');
            const message = input.value.trim();
            const model = document.getElementById('chatModel').value;
            
            if (!message) return;

            const messagesContainer = document.getElementById('chatbotMessages');
            
            messagesContainer.innerHTML += `
                <div class="chat-message user-message">
                    <div class="message-avatar">üë§</div>
                    <div class="message-content"><p>${message}</p></div>
                </div>
            `;
            
            input.value = '';
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
            
            const loadingId = Date.now();
            messagesContainer.innerHTML += `
                <div class="chat-message bot-message" id="loading-${loadingId}">
                    <div class="message-avatar">ü§ñ</div>
                    <div class="message-content"><p>üîÑ Thinking...</p></div>
                </div>
            `;
            messagesContainer.scrollTop = messagesContainer.scrollHeight;

            try {
                let response;
                
                if (model === 'pollinations') {
                    response = await fetch(`https://text.pollinations.ai/${encodeURIComponent(message)}`);
                    const text = await response.text();
                    showBotResponse(text, loadingId);
                } else {
                    // Fallback to Pollinations for other models in this static version
                    response = await fetch(`https://text.pollinations.ai/${encodeURIComponent(message)}`);
                    const text = await response.text();
                    showBotResponse(text, loadingId);
                }
                
            } catch (error) {
                console.error('Chat error:', error);
                showBotResponse('Sorry, I encountered an error. Please try again.', loadingId);
            }
        }

        function showBotResponse(text, loadingId) {
            const loadingElement = document.getElementById(`loading-${loadingId}`);
            if (loadingElement) {
                loadingElement.innerHTML = `
                    <div class="message-avatar">ü§ñ</div>
                    <div class="message-content"><p>${text}</p></div>
                `;
                loadingElement.id = '';
            }
            
            const messagesContainer = document.getElementById('chatbotMessages');
            messagesContainer.scrollTop = messagesContainer.scrollHeight;
        }
